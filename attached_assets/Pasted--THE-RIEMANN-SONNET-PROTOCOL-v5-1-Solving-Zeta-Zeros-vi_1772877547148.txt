def poetic_constraint_proof():

      - Pentameter: 5 feet = 10 syllables
      - Stress pattern: 0101010101
      - Spondee: 11 → Re(s) > 1/2
      - Pyrrhic: 00 → Re(s) < 1/2

    iambic = [0, 1] * 5  # da-DUM da-DUM da-DUM da-DUM da-DUM

                      for i in range(len(iambic)-1))
    
    # Check: no consecutive unstressed (pyrrhic)

    # Deviation from critical line

    return {

        'has_pyrrhic': has_pyrrhic,
