    // Exponential height-falloff attenuation
    float fog_factor = exp(-view_height * 0.002) * (1.0 - exp(-distance * fog_density));
    fog_factor = clamp(fog_factor, 0.0, 1.0);
    return mix(color, u_fog_color_density.rgb, fog_factor);
}
void main() {

    vec2 tex_coords = v_world_pos.xz * 0.001; // Scale factor matching terrain boundary
    vec4 packed_data = texture(u_packed_surface, tex_coords);
    
    float macro_ao = packed_data.r;
    float roughness = packed_data.a;

    vec3 normal_t;
    normal_t.x = packed_data.g * 2.0 - 1.0;
    normal_t.z = packed_data.b * 2.0 - 1.0;
    normal_t.y = sqrt(max(0.0, 1.0 - (normal_t.x * normal_t.x + normal_t.z * normal_t.z)));

## Task 3: Billion-Tree Instancing & The Memory Pipeline
To fit 250,000 trees into an 800KB network payload, we abandon standard JSON serialization entirely. Instead, we pack each tree instance into a tight 24-bit bit-packed layout (3 bytes per instance).
## Binary Blueprint (.jacoforest)

* X-Coordinate Offset: 8 bits (0–255 mapped to patch boundaries)
* Z-Coordinate Offset: 8 bits (0–255 mapped to patch boundaries)
* Scale Factor: 4 bits (Quantized to 16 baseline variants)
* Y-Rotation Angle: 4 bits (Quantized steps of $22.5^\circ$)

## Production Parser & WebGL Buffer Pipeline

/**
 * Zero-Allocation Binary Stream Parser 
 * Executed inside an isolated Web Worker thread
 */export function parseJacoForest(arrayBuffer: ArrayBuffer, patchOffset: [number, number]): Float32Array {
  const srcView = new DataView(arrayBuffer);
  const count = arrayBuffer.byteLength / 3; // 3 bytes per packed instance
  
  // Target format: 4 floats per instance for raw instanced VBO [X, Y_elevation, Z, packed_scale_rot]
  const vboData = new Float32Array(count * 4);
  
  for (let i = 0; i < count; i++) {
    const byteOffset = i * 3;
    
    // Pull packed byte segments sequentially
    const posX = srcView.getUint8(byteOffset);
    const posZ = srcView.getUint8(byteOffset + 1);
    const flags = srcView.getUint8(byteOffset + 2);
    
    // Extract quantized values via bitmasks
    const scaleBits = (flags >> 4) & 0x0F;
    const rotBits = flags & 0x0F;
    
    // Map integers back to normalized engine space
    const worldX = patchOffset[0] + (posX / 255.0) * 256.0;
    const worldZ = patchOffset[1] + (posZ / 255.0) * 256.0;
    const scale = 0.5 + (scaleBits / 15.0) * 2.0; // scale bounds: 0.5 - 2.5
    const rotationRad = (rotBits / 15.0) * Math.PI * 2.0;
    
    // Store attributes sequentially inside interleaved layout array
    const destIdx = i * 4;
    vboData[destIdx]     = worldX;
    vboData[destIdx + 1] = 0.0; // Handled directly via Vertex Shader heightmap sampling
    vboData[destIdx + 2] = worldZ;
    
    // Pack scale and rotation float attributes into a single 32-bit register to conserve slot constraints
    vboData[destIdx + 3] = scale * 1000.0 + rotationRad;
  }
  
  return vboData;
}
/**
 * GPU Pipeline Binding Engine (Main Thread Execution)
 */export function bindInstancedTrees(gl: WebGL2RenderingContext, vboData: Float32Array) {
  const treeVBO = gl.createBuffer();
  gl.bindBuffer(gl.ARRAY_BUFFER, treeVBO);
  gl.bufferData(gl.ARRAY_BUFFER, vboData, gl.STATIC_DRAW);
  
  // Setup stride offsets to stream coordinates directly into GPU instance registers
  const stride = 4 * 4; // 4 attributes * 4 bytes per float
  
  // Instance Attribute Location: position data pointer
  gl.enableVertexAttribArray(2);
  gl.vertexAttribPointer(2, 4, gl.FLOAT, false, stride, 0);
  gl.vertexAttribDivisor(2, 1); // Tells the GPU to increment layout pointer per-instance, not per-vertex
}
