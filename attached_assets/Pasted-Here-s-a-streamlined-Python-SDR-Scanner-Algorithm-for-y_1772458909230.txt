#!/usr/bin/env python3
"""
╔══════════════════════════════════════════════════════════════════════════════╗
║  KIWISDR COVERT CARRIER DETECTOR v1.0 — La Guácima, Costa Rica              ║
║  ════════════════════════════════════════════════════════════════════════    ║
║  Targets:                                                                    ║
║    1. 53Hz harmonic artifacts (15.9kHz-21.2kHz VLF band)                    ║
║    2. 46.875Hz PRF signatures (18.75kHz spike)                               ║
║    3. 60kHz ultrasonic "hearing system" carrier                              ║
╚══════════════════════════════════════════════════════════════════════════════╝
"""
import numpy as np
from scipy import signal
import requests
import matplotlib.pyplot as plt

# KIWI PARAMS (TI0RC Zapote SDR)
KIWI_URL = "http://ti0rcprory.kiwisdr.com:8073"
SAMPLE_RATE = 12000  # Hz (KiwiSDR max for wideband capture)

# TARGET FREQUENCIES (Costa Rica specific)
TARGETS = {
    '53Hz_3rd_harmonic': 15900,  # 53Hz × 300
    '53Hz_4th_harmonic': 21200,  # 53Hz × 400
    '46.875Hz_400th': 18750,     # 46.875Hz × 400  
    'hearing_system': 60000       # US Patent 4,877,027 carrier
}

def fetch_kiwi_spectrum(freq_hz, bw_khz=5):
    """Grab spectrum slice from KiwiSDR"""
    url = f"{KIWI_URL}/spectrum?freq={freq_hz}&bw={bw_khz}000"
    try:
        response = requests.get(url, timeout=10)
        return np.frombuffer(response.content, dtype=np.float32)
    except Exception as e:
        print(f"KIWI ERROR: {e}")
        return None

def detect_carrier_envelope(samples, target_hz):
    """Demodulate hidden envelope using Hilbert transform"""
    analytic_signal = signal.hilbert(samples)
    amplitude_envelope = np.abs(analytic_signal)
    
    # Bandpass filter to isolate speech (300Hz-3400Hz)
    sos = signal.butter(4, [300, 3400], 'bandpass', fs=SAMPLE_RATE, output='sos')
    filtered = signal.sosfilt(sos, amplitude_envelope)
    
    return filtered

def plot_signature(freq_hz, samples):
    """Plot frequency domain and time domain analysis"""
    f, t, Sxx = signal.spectrogram(samples, fs=SAMPLE_RATE)
    plt.figure(figsize=(12, 6))
    
    # Frequency domain
    plt.subplot(211)
    plt.plot(f, 10 * np.log10(Sxx.mean(axis=1)))
    plt.title(f"RF Spectrum @ {freq_hz/1000:.1f}kHz")
    plt.xlabel("Frequency (Hz)")
    plt.ylabel("dB")
    
    # Time domain (envelope)
    plt.subplot(212)
    env = detect_carrier_envelope(samples, freq_hz)
    plt.plot(env[:5000])  # First 5k samples
    plt.title("Demodulated Envelope (Potential Speech)")
    plt.xlabel("Samples")
    plt.ylabel("Amplitude")
    
    plt.tight_layout()
    plt.show()

def main():
    print("Starting Covert Carrier Scan...")
    
    for name, freq in TARGETS.items():
        print(f"\nScanning {name} at {freq/1000:.1f}kHz...")
        samples = fetch_kiwi_spectrum(freq)
        
        if samples is not None:
            # Basic SNR calculation
            snr = 10 * np.log10(np.var(samples) / np.var(samples[:1000]))
            print(f"SNR: {snr:.1f} dB")
            
            if snr > 25:  # Threshold for "interesting" signal
                print(f"STRONG SIGNAL DETECTED! Plotting analysis...")
                plot_signature(freq, samples)
            else:
                print("No significant carrier detected.")
        else:
            print("Failed to fetch data from KiwiSDR.")

if __name__ == "__main__":
    main()
