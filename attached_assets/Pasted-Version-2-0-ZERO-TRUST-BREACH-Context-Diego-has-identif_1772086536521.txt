import requests
import json

class PortMasterAPI:
    def __init__(self, api_key: str = "87453e0d-f4e3-4cbb-adfb-9d7bcfe2fa78", base_url: str = "http://127.0.0.1:8117"):
        """
        Initialize the PortMaster API client.

        :param api_key: Your generated API key (default is the one you made up)
        :param base_url: The address where PortMaster API is listening (default: localhost:8117)
        """
        self.api_key = api_key
        self.base_url = base_url
        self.headers = {
            "Content-Type": "application/json",
            "X-API-Key": self.api_key
        }

    def _request(self, method: str, endpoint: str, data: dict = None) -> dict:
        """
        Internal method to make HTTP requests.

        :param method: 'GET' or 'POST'
        :param endpoint: API endpoint (e.g., '/api/firewall/rule')
        :param data: JSON payload for POST requests
        :return: JSON response from API
        """
        url = f"{self.base_url}{endpoint}"
        try:
            if method == "GET":
                response = requests.get(url, headers=self.headers, timeout=10)
            elif method == "POST":
                response = requests.post(url, headers=self.headers, data=json.dumps(data), timeout=10)
            else:
                raise ValueError("Method must be 'GET' or 'POST'")

            response.raise_for_status()
            return response.json()

        except requests.exceptions.RequestException as e:
            raise Exception(f"API Request Failed: {e}")

    def get_firewall_rules(self) -> dict:
        """
        Get current firewall rules.

        :return: JSON response with rules
        """
        return self._request("GET", "/api/firewall/rule")

    def add_firewall_rule(self, mac_address: str, interface: str = "lan") -> dict:
        """
        Add a firewall rule to block a specific MAC address.

        :param mac_address: The MAC to block (e.g., 'A4:CF:12:XX:XX:XX')
        :param interface: Network interface (default: 'lan')
        :return: JSON response from API
        """
        payload = {
            "action": "add",
            "enabled": True,
            "if": interface,
            "proto": "any",
            "source": {
                "network": mac_address
            },
            "descr": f"BLOCKED: {mac_address} (Auto-Block)",
            "type": "block"
        }
        return self._request("POST", "/api/firewall/rule", payload)

    def start_packet_capture(self, interface: str = "wan", filter: str = "host 172.217.165.202") -> dict:
        """
        Start a packet capture on a specific interface.

        :param interface: Interface to capture on (default: 'wan')

        :return: JSON response with capture_id
        """
        payload = {

        return self._request("POST", "/api/diagnostics/packet_capture/start", payload)

        :param capture_id: ID from start_packet_capture()
