// Superposition rendering shader
export const SuperpositionShader = {
  uniforms: {
    time: { value: 0 },
    superpositionStates: { value: new Array(10).fill(new THREE.Vector3()) },
    amplitudes: { value: new Array(10).fill(0) },
    phases: { value: new Array(10).fill(0) },
    decoherenceTime: { value: 1.273 }, // κ-scaled
    measurementStrength: { value: 0.0 }
  },
  
  vertexShader: `
    uniform float time;
    uniform vec3 superpositionStates[10];
    uniform float amplitudes[10];
    uniform float phases[10];
    
    varying vec3 vPosition;
    varying vec3 vNormal;
    varying vec4 vQuantumState;
    
    void main() {
      // Quantum superposition: |ψ⟩ = Σ c_i |i⟩
      vec3 superposedPosition = vec3(0.0);
      float totalAmplitude = 0.0;
      
      for(int i = 0; i < 10; i++) {
        float amplitude = amplitudes[i];
        float phase = phases[i] + time * 0.5;
        vec3 statePosition = superpositionStates[i];
        
        // Complex coefficient: c = amplitude * exp(i*phase)
        vec2 coefficient = amplitude * vec2(cos(phase), sin(phase));
        
        // Interference term: |c|²
        float interference = dot(coefficient, coefficient);
        superposedPosition += interference * statePosition;
        totalAmplitude += interference;
      }
      
      // Normalize
      if(totalAmplitude > 0.0) {
        superposedPosition /= totalAmplitude;
      }
      
      vec4 worldPosition = modelMatrix * vec4(position + superposedPosition * 0.1, 1.0);
      vPosition = worldPosition.xyz;
      vNormal = normalMatrix * normal;
      
      // Store quantum state info for fragment shader
      vQuantumState = vec4(superposedPosition, totalAmplitude);
      
      gl_Position = projectionMatrix * viewMatrix * worldPosition;
    }
  `,
  
  fragmentShader: `
    uniform float time;
    uniform float decoherenceTime;
    uniform float measurementStrength;
    
    varying vec3 vPosition;
    varying vec3 vNormal;
    varying vec4 vQuantumState;
    
    #define M_PI 3.14159265359
    
    // Born rule: probability = |ψ|²
    float bornRuleProbability(vec4 quantumState) {
      return dot(quantumState.xyz, quantumState.xyz);
    }
    
    // Wigner function (quasi-probability distribution)
    float wignerFunction(vec2 phaseSpace) {
      float q = phaseSpace.x;
      float p = phaseSpace.y;
      return exp(-q*q - p*p) / M_PI;
    }
    
    void main() {
      // Compute probability density
      float probability = bornRuleProbability(vQuantumState);
      
      // Apply measurement (wave function collapse simulation)
      float collapsed = step(0.5, probability);
      
      // Decoherence effect
      float coherence = exp(-time / decoherenceTime);
      float alpha = mix(probability, collapsed, 1.0 - coherence) * measurementStrength 
                  + probability * (1.0 - measurementStrength);
      
      // Quantum interference colors (based on phase)
      vec3 quantumColor = vec3(
        sin(vQuantumState.x * 10.0 + time) * 0.5 + 0.5,
        cos(vQuantumState.y * 10.0 + time * 0.7) * 0.5 + 0.5,
        sin(vQuantumState.z * 10.0 + time * 0.3) * 0.5 + 0.5
      );
      
      // Lighting
      vec3 lightDir = normalize(vec3(1.0, 1.0, 1.0));
      float diffuse = max(dot(vNormal, lightDir), 0.0);
      
      // Final color with quantum effects
      vec3 finalColor = quantumColor * (0.3 + 0.7 * diffuse) * alpha;
      
      gl_FragColor = vec4(finalColor, 1.0);
    }
  `
};

// Entanglement visualization shader
export const EntanglementShader = {
  uniforms: {
    time: { value: 0 },
    partnerPosition: { value: new THREE.Vector3() },
    entanglementStrength: { value: 1.0 },
    bellViolation: { value: 0.0 }
  },
  
  vertexShader: `
    uniform float time;
    uniform vec3 partnerPosition;
    uniform float entanglementStrength;
    
    varying vec3 vEntanglement;
    varying float vStrength;
    varying vec3 vWorldPosition;
    
    void main() {
      vEntanglement = partnerPosition - position;
      vStrength = entanglementStrength;
      
      // Entangled motion (spooky action at a distance)
      vec3 entangledOffset = normalize(vEntanglement) 
                           * sin(time * entanglementStrength * 2.0) * 0.05;
      
      vec4 worldPosition = modelMatrix * vec4(position + entangledOffset, 1.0);
      vWorldPosition = worldPosition.xyz;
      
      gl_Position = projectionMatrix * viewMatrix * worldPosition;
    }
  `,
  
  fragmentShader: `
    uniform float time;
    uniform float bellViolation;
    
    varying vec3 vEntanglement;
    varying float vStrength;
    varying vec3 vWorldPosition;
    
    void main() {
      // Entanglement visualization
      float distance = length(vEntanglement);
      float pulse = sin(distance * 10.0 - time * 3.0) * 0.5 + 0.5;
      
      // Quantum correlation colors
      // Tsirelson bound: 2√2 ≈ 2.828
      vec3 color = mix(
        vec3(0.0, 1.0, 1.0), // Separable (cyan)
        vec3(1.0, 0.0, 1.0), // Entangled (magenta)
        smoothstep(2.0, 2.828, bellViolation)
      );
      
      color *= pulse * vStrength;
      
      gl_FragColor = vec4(color, vStrength);
    }
  `
};

// Quantum tunneling shader
export const TunnelingShader = {
  uniforms: {
    time: { value: 0 },
    barrierHeight: { value: 1.0 },
    barrierWidth: { value: 0.1 },
    particleEnergy: { value: 0.5 },
    barrierStart: { value: 0.4 },
    barrierEnd: { value: 0.6 }
  },
  
  vertexShader: `
    varying vec2 vUv;
    varying vec3 vPosition;
    
    void main() {
      vUv = uv;
      vPosition = position;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  
  fragmentShader: `
    uniform float time;
    uniform float barrierHeight;
    uniform float barrierWidth;
    uniform float particleEnergy;
    uniform float barrierStart;
    uniform float barrierEnd;
    
    varying vec2 vUv;
    varying vec3 vPosition;
    
    #define M_PI 3.14159265359
    
    float computeTransmission(float V0, float E, float d) {
      if (E >= V0) return 1.0;
      
      float kappa = sqrt(2.0 * (V0 - E));
      float arg = kappa * d;
      
      if (arg > 20.0) return exp(-2.0 * arg);
      
      float sinhTerm = sinh(arg);
      float denom = 1.0 + V0 * V0 * sinhTerm * sinhTerm 
                  / (4.0 * E * (V0 - E));
      return 1.0 / denom;
    }
    
    void main() {
      float x = vUv.x;
      float y = vUv.y - 0.5;
      
      // Wave packet parameters
      float k = sqrt(2.0 * particleEnergy) * 20.0;
      float sigma = 0.1;
      float x0 = mod(time * 0.1, 1.0);
      
      // Gaussian wave packet
      float envelope = exp(-pow(x - x0, 2.0) / (2.0 * sigma * sigma));
      float wave = cos(k * x - time * 5.0);
      
      // Barrier region
      bool inBarrier = x >= barrierStart && x <= barrierEnd;
      
      // Transmission coefficient
      float T = computeTransmission(barrierHeight, particleEnergy, barrierWidth);
      
      // Wave function
      float psi;
      if (x < barrierStart) {
        // Incident + reflected
        psi = envelope * (wave + (1.0 - T) * cos(-k * x - time * 5.0));
      } else if (inBarrier) {
        // Evanescent wave
        float kappa = sqrt(2.0 * (barrierHeight - particleEnergy));
        psi = envelope * T * exp(-kappa * (x - barrierStart));
      } else {
        // Transmitted
        psi = envelope * T * wave;
      }
      
      // Probability density |ψ|²
      float probability = psi * psi;
      
      // Colors
      vec3 incidentColor = vec3(0.0, 0.5, 1.0);  // Blue for incident
      vec3 barrierColor = vec3(1.0, 0.3, 0.0);   // Orange for barrier
      vec3 transmittedColor = vec3(0.0, 1.0, 0.5); // Green for transmitted
      
      vec3 color;
      if (x < barrierStart) {
        color = incidentColor;
      } else if (inBarrier) {
        color = barrierColor;
      } else {
        color = transmittedColor;
      }
      
      // Draw wave function
      float waveY = probability * 2.0;
      float line = smoothstep(0.02, 0.0, abs(y - waveY));
      
      // Barrier visualization
      float barrier = inBarrier ? 0.3 : 0.0;
      
      vec3 finalColor = color * (line + barrier * 0.5);
      
      gl_FragColor = vec4(finalColor, 1.0);
    }
  `
};

// Wave function visualization shader
export const WaveFunctionShader = {
  uniforms: {
    time: { value: 0 },
    waveFunctionTexture: { value: null },
    potentialTexture: { value: null },
    showProbability: { value: true },
    showPhase: { value: false }
  },
  
  vertexShader: `
    varying vec2 vUv;
    
    void main() {
      vUv = uv;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  
  fragmentShader: `
    uniform float time;
    uniform sampler2D waveFunctionTexture;
    uniform sampler2D potentialTexture;
    uniform bool showProbability;
    uniform bool showPhase;
    
    varying vec2 vUv;
    
    #define M_PI 3.14159265359
    
    vec3 hsv2rgb(vec3 c) {
      vec4 K = vec4(1.0, 2.0 / 3.0, 1.0 / 3.0, 3.0);
      vec3 p = abs(fract(c.xxx + K.xyz) * 6.0 - K.www);
      return c.z * mix(K.xxx, clamp(p - K.xxx, 0.0, 1.0), c.y);
    }
    
    void main() {
      vec4 psi = texture2D(waveFunctionTexture, vUv);
      float potential = texture2D(potentialTexture, vUv).r;
      
      // Complex wave function: psi.r = real, psi.g = imag
      float realPart = psi.r;
      float imagPart = psi.g;
      
      // Probability density |ψ|²
      float probability = realPart * realPart + imagPart * imagPart;
      
      // Phase angle
      float phase = atan(imagPart, realPart);
      
      vec3 color;
      
      if (showProbability && !showPhase) {
        // Probability density visualization
        color = vec3(probability) * vec3(0.2, 0.5, 1.0);
      } else if (showPhase && !showProbability) {
        // Phase visualization (HSV color wheel)
        float hue = (phase + M_PI) / (2.0 * M_PI);
        color = hsv2rgb(vec3(hue, 1.0, 1.0));
      } else {
        // Combined: brightness = probability, hue = phase
        float hue = (phase + M_PI) / (2.0 * M_PI);
        color = hsv2rgb(vec3(hue, 1.0, sqrt(probability)));
      }
      
      // Add potential barrier visualization
      color += vec3(potential * 0.3, potential * 0.1, 0.0);
      
      gl_FragColor = vec4(color, 1.0);
    }
  `
};

export class QuantumRenderer {
  public renderer: THREE.WebGLRenderer;
  public scene: THREE.Scene;
  public camera: THREE.PerspectiveCamera;
  public quantumState: QuantumState;
  public fieldEngine: QuantumFieldEngine;
  public gravityEngine: QuantumGravityEngine;
  
  private clock: THREE.Clock;
  private superpositionMaterial: THREE.ShaderMaterial;
  private entanglementMaterial: THREE.ShaderMaterial;
  private tunnelingMaterial: THREE.ShaderMaterial;
  private waveFunctionMaterial: THREE.ShaderMaterial;
  private waveFunctionTexture: THREE.DataTexture;
  private potentialTexture: THREE.DataTexture;
  
  private quantumObjects: THREE.Mesh[] = [];
  private entangledPairs: Array<{ a: THREE.Mesh; b: THREE.Mesh; strength: number }> = [];

  constructor(container: HTMLElement, width?: number, height?: number) {
    this.clock = new THREE.Clock();
    
    // Initialize renderer
    this.renderer = new THREE.WebGLRenderer({ 
      antialias: true,
      alpha: true
    });
    this.renderer.setSize(width || container.clientWidth, height || container.clientHeight);
    this.renderer.setPixelRatio(window.devicePixelRatio);
    container.appendChild(this.renderer.domElement);
    
    // Scene setup
    this.scene = new THREE.Scene();
    this.scene.background = new THREE.Color(0x050510);
    
    // Camera
    this.camera = new THREE.PerspectiveCamera(
      75,
      (width || container.clientWidth) / (height || container.clientHeight),
      0.001,
      1000
    );
    this.camera.position.z = 5;
    
    // Quantum engines
    this.quantumState = new QuantumState(16);
    this.fieldEngine = new QuantumFieldEngine(64);
    this.gravityEngine = new QuantumGravityEngine(200);
    
    // Create materials
    this.superpositionMaterial = new THREE.ShaderMaterial(SuperpositionShader);
    this.entanglementMaterial = new THREE.ShaderMaterial(EntanglementShader);
    this.tunnelingMaterial = new THREE.ShaderMaterial(TunnelingShader);
    
    // Wave function textures
    this.waveFunctionTexture = this.createWaveFunctionTexture();
    this.potentialTexture = this.createPotentialTexture();
    
    this.waveFunctionMaterial = new THREE.ShaderMaterial({
      ...WaveFunctionShader,
      uniforms: {
        ...WaveFunctionShader.uniforms,
        waveFunctionTexture: { value: this.waveFunctionTexture },
        potentialTexture: { value: this.potentialTexture }
      }
    });
    
    // Lighting
    this.setupLighting();
    
    // Initialize quantum effects
    this.initQuantumEffects();
  }

  private setupLighting(): void {
    const ambientLight = new THREE.AmbientLight(0x404040, 0.5);
    this.scene.add(ambientLight);
    
    const directionalLight = new THREE.DirectionalLight(0xffffff, 1);
    directionalLight.position.set(5, 5, 5);
    this.scene.add(directionalLight);
    
    const pointLight = new THREE.PointLight(0x8888ff, 1, 10);
    pointLight.position.set(0, 2, 0);
    this.scene.add(pointLight);
  }

  private createWaveFunctionTexture(): THREE.DataTexture {
    const size = 128;
    const data = new Float32Array(size * size * 4);
    
    // Initialize with Gaussian wave packet
    const wf = new WaveFunction2D(size);
    wf.initializeGaussianWavePacket(0.3, 0.5, 5, 0, 0.05);
    
    for (let i = 0; i < size * size; i++) {
      const psi = wf.psi[i];
      data[i * 4] = psi.real;
      data[i * 4 + 1] = psi.imag;
      data[i * 4 + 2] = 0;
      data[i * 4 + 3] = 1;
    }
    
    const texture = new THREE.DataTexture(
      data, size, size,
      THREE.RGBAFormat,
      THREE.FloatType
    );
    texture.needsUpdate = true;
    
    return texture;
  }

  private createPotentialTexture(): THREE.DataTexture {
    const size = 128;
    const data = new Float32Array(size * size * 4);
    
    // Double slit potential
    const wallPos = 0.4;
    const wallWidth = 0.05;
    const slitWidth = 0.05;
    const slitSep = 0.15;
    
    for (let y = 0; y < size; y++) {
      for (let x = 0; x < size; x++) {
        const xNorm = x / size;
        const yNorm = y / size;
        const idx = (y * size + x) * 4;
        
        let potential = 0;
        
        if (xNorm >= wallPos && xNorm <= wallPos + wallWidth) {
          const dy1 = Math.abs(yNorm - (0.5 - slitSep / 2));
          const dy2 = Math.abs(yNorm - (0.5 + slitSep / 2));
          
          if (dy1 > slitWidth / 2 && dy2 > slitWidth / 2) {
            potential = 100;
          }
        }
        
        data[idx] = potential;
        data[idx + 1] = 0;
        data[idx + 2] = 0;
        data[idx + 3] = 1;
      }
    }
    
    const texture = new THREE.DataTexture(
      data, size, size,
      THREE.RGBAFormat,
      THREE.FloatType
    );
    texture.needsUpdate = true;
    
    return texture;
  }

  private initQuantumEffects(): void {
    // 1. Quantum superposition sphere
    this.createSuperpositionSphere();
    
    // 2. Entangled particles
    this.createEntangledPairs(3);
    
    // 3. Tunneling visualization plane
    this.createTunnelingPlane();
    
    // 4. Wave function display
    this.createWaveFunctionDisplay();
    
    // 5. Causal set visualization
    this.createCausalSetVisualization();
  }

  private createSuperpositionSphere(): void {
    const geometry = new THREE.SphereGeometry(1, 64, 64);
    const mesh = new THREE.Mesh(geometry, this.superpositionMaterial.clone());
    mesh.position.set(-3, 0, 0);
    this.scene.add(mesh);
    this.quantumObjects.push(mesh);
    
    // Initialize superposition states
    const material = mesh.material as THREE.ShaderMaterial;
    for (let i = 0; i < 10; i++) {
      const theta = Math.random() * Math.PI;
      const phi = Math.random() * 2 * Math.PI;
      material.uniforms.superpositionStates.value[i] = new THREE.Vector3(
        Math.sin(theta) * Math.cos(phi),
        Math.sin(theta) * Math.sin(phi),
        Math.cos(theta)
      );
      material.uniforms.amplitudes.value[i] = Math.random();
      material.uniforms.phases.value[i] = Math.random() * 2 * Math.PI;
    }
  }

  private createEntangledPairs(numPairs: number): void {
    for (let i = 0; i < numPairs; i++) {
      const geometry = new THREE.IcosahedronGeometry(0.3, 2);
      
      const materialA = this.entanglementMaterial.clone();
      const materialB = this.entanglementMaterial.clone();
      
      const meshA = new THREE.Mesh(geometry, materialA);
      const meshB = new THREE.Mesh(geometry, materialB);
      
      const angle = (i / numPairs) * Math.PI * 2;
      const radius = 2;
      
      meshA.position.set(Math.cos(angle) * radius, 1.5, Math.sin(angle) * radius);
      meshB.position.set(-Math.cos(angle) * radius, 1.5, -Math.sin(angle) * radius);
      
      this.scene.add(meshA);
      this.scene.add(meshB);
      
      this.entangledPairs.push({
        a: meshA,
        b: meshB,
        strength: 0.8 + Math.random() * 0.2
      });
      
      // Connection line
      const lineGeometry = new THREE.BufferGeometry().setFromPoints([
        meshA.position,
        meshB.position
      ]);
      const lineMaterial = new THREE.LineBasicMaterial({
        color: 0xff00ff,
        transparent: true,
        opacity: 0.3
      });
      const line = new THREE.Line(lineGeometry, lineMaterial);
      this.scene.add(line);
    }
  }

  private createTunnelingPlane(): void {
    const geometry = new THREE.PlaneGeometry(4, 2);
    const mesh = new THREE.Mesh(geometry, this.tunnelingMaterial);
    mesh.position.set(3, 0, 0);
    this.scene.add(mesh);
  }

  private createWaveFunctionDisplay(): void {
    const geometry = new THREE.PlaneGeometry(3, 3);
    const mesh = new THREE.Mesh(geometry, this.waveFunctionMaterial);
    mesh.position.set(0, -2.5, 0);
    mesh.rotation.x = -Math.PI / 4;
    this.scene.add(mesh);
  }

  private createCausalSetVisualization(): void {
    const data = this.gravityEngine.getVisualizationData();
    
    // Nodes as small spheres
    const nodeGeometry = new THREE.SphereGeometry(0.02, 8, 8);
    const nodeMaterial = new THREE.MeshBasicMaterial({ color: 0x00ffff });
    
    const nodeGroup = new THREE.Group();
    
    data.nodes.forEach(node => {
      const mesh = new THREE.Mesh(nodeGeometry, nodeMaterial.clone());
      mesh.position.set(
        node.x * 2,
        node.t * 2 - 1,
        node.y * 2
      );
      
      // Color by curvature
      const curvatureColor = new THREE.Color();
      curvatureColor.setHSL(0.6 - node.curvature * 0.1, 1, 0.5);
      (mesh.material as THREE.MeshBasicMaterial).color = curvatureColor;
      
      nodeGroup.add(mesh);
    });
    
    // Causal links
    const lineMaterial = new THREE.LineBasicMaterial({
      color: 0x444488,
      transparent: true,
      opacity: 0.2
    });
    
    data.links.slice(0, 500).forEach(link => {
      const from = data.nodes[link.from];
      const to = data.nodes[link.to];
      
      const points = [
        new THREE.Vector3(from.x * 2, from.t * 2 - 1, from.y * 2),
        new THREE.Vector3(to.x * 2, to.t * 2 - 1, to.y * 2)
      ];
      
      const geometry = new THREE.BufferGeometry().setFromPoints(points);
      const line = new THREE.Line(geometry, lineMaterial);
      nodeGroup.add(line);
    });
    
    nodeGroup.position.set(0, 2, -2);
    nodeGroup.scale.set(0.5, 0.5, 0.5);
    this.scene.add(nodeGroup);
  }

  update(): void {
    const time = this.clock.getElapsedTime();
    const delta = this.clock.getDelta();
    
    // Update quantum state
    this.quantumState.evolve(delta * 0.001, {
      terms: [{ 
        operator: this.quantumState.identityMatrix(), 
        coefficient: κ 
      }],
      dimension: this.quantumState.dimensions
    });
    
    // Update field engine
    this.fieldEngine.evolveField(delta * 0.01);
    
    // Update shader uniforms
    this.updateShaderUniforms(time);
    
    // Update entangled pairs
    this.updateEntangledPairs(time);
    
    // Update wave function texture
    this.updateWaveFunctionTexture(time);
    
    // Render
    this.renderer.render(this.scene, this.camera);
  }

  private updateShaderUniforms(time: number): void {
    // Superposition material
    this.quantumObjects.forEach(obj => {
      const material = obj.material as THREE.ShaderMaterial;
      if (material.uniforms.time) {
        material.uniforms.time.value = time;
      }
    });
    
    // Tunneling material
    this.tunnelingMaterial.uniforms.time.value = time;
    this.tunnelingMaterial.uniforms.particleEnergy.value = 
      0.3 + 0.2 * Math.sin(time * 0.5);
    
    // Wave function material
    this.waveFunctionMaterial.uniforms.time.value = time;
  }

  private updateEntangledPairs(time: number): void {
    this.entangledPairs.forEach(pair => {
      const materialA = pair.a.material as THREE.ShaderMaterial;
      const materialB = pair.b.material as THREE.ShaderMaterial;
      
      materialA.uniforms.time.value = time;
      materialB.uniforms.time.value = time;
      
      materialA.uniforms.partnerPosition.value = pair.b.position;
      materialB.uniforms.partnerPosition.value = pair.a.position;
      
      materialA.uniforms.entanglementStrength.value = pair.strength;
      materialB.uniforms.entanglementStrength.value = pair.strength;
      
      // Bell inequality violation (2√2 for maximally entangled)
      const violation = 2 * Math.sqrt(2) * pair.strength;
      materialA.uniforms.bellViolation.value = violation;
      materialB.uniforms.bellViolation.value = violation;
    });
  }

  private updateWaveFunctionTexture(time: number): void {
    const size = 128;
    const data = this.waveFunctionTexture.image.data as Float32Array;
    
    // Simple wave packet evolution (for visualization)
    const k = 10;
    const x0 = 0.3 + 0.3 * Math.sin(time * 0.2);
    const sigma = 0.08;
    
    for (let y = 0; y < size; y++) {
      for (let x = 0; x < size; x++) {
        const idx = (y * size + x) * 4;
        const xNorm = x / size;
        const yNorm = y / size;
        
        // Check if in barrier
        const inBarrier = xNorm >= 0.4 && xNorm <= 0.45;
        const dy1 = Math.abs(yNorm - 0.35);
        const dy2 = Math.abs(yNorm - 0.65);
        const inSlit = dy1 < 0.05 || dy2 < 0.05;
        
        if (inBarrier && !inSlit) {
          // Barrier region
          data[idx] = 0;
          data[idx + 1] = 0;
        } else {
          // Wave packet
          const dx = xNorm - x0;
          const dy = yNorm - 0.5;
          const envelope = Math.exp(-(dx * dx + dy * dy) / (2 * sigma * sigma));
          const phase = k * xNorm - time * 5;
          
          data[idx] = envelope * Math.cos(phase);
          data[idx + 1] = envelope * Math.sin(phase);
        }
      }
    }
    
    this.waveFunctionTexture.needsUpdate = true;
  }

  resize(width: number, height: number): void {
    this.camera.aspect = width / height;
    this.camera.updateProjectionMatrix();
    this.renderer.setSize(width, height);
  }

  dispose(): void {
    this.renderer.dispose();
    this.scene.traverse((obj: THREE.Object3D) => {
      if (obj instanceof THREE.Mesh) {
        obj.geometry.dispose();
        if (obj.material instanceof THREE.Material) {
          obj.material.dispose();
        }
      }
    });
  }

  // Get quantum metrics for UI
  getQuantumMetrics(): {
    coherence: number;
    entanglement: number;
    purity: number;
    vonNeumannEntropy: number;
    fieldEnergy: number;
    spacetimeDimension: number;
    averageCurvature: number;
  } {
    return {
      coherence: this.quantumState.computeCoherence(),
      entanglement: this.quantumState.entanglementEntropy,
      purity: this.quantumState.computePurity(),
      vonNeumannEntropy: this.quantumState.computeVonNeumannEntropy(),
      fieldEnergy: this.fieldEngine.getTotalEnergy(),
      spacetimeDimension: this.gravityEngine.getDimension(),
      averageCurvature: this.gravityEngine.getAverageCurvature()
    };
  }
}

export const DecoherenceShader = {
  uniforms: {
    tDiffuse: { value: null },
    time: { value: 0 },
    decoherenceRate: { value: 0.1 },
    measurementStrength: { value: 0.0 },
    resolution: { value: new THREE.Vector2(1, 1) }
  },
  
  vertexShader: `
    varying vec2 vUv;
    void main() {
      vUv = uv;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  
  fragmentShader: `
    uniform sampler2D tDiffuse;
    uniform float time;
    uniform float decoherenceRate;
    uniform float measurementStrength;
    uniform vec2 resolution;
    
    varying vec2 vUv;
    
    // Pseudo-random for GPU
    float random(vec2 st) {
      return fract(sin(dot(st.xy, vec2(12.9898, 78.233))) * 43758.5453123);
    }
    
    // Lindblad-inspired decoherence simulation
    vec4 lindbladEvolution(vec4 rho, float dt) {
      // Simplified decoherence: off-diagonal elements decay exponentially
      float decay = exp(-decoherenceRate * dt);
      
      // Treat RGB as diagonal + off-diagonal density matrix elements
      vec3 diagonal = rho.rgb * (0.5 + 0.5 * decay);
      
      // Add quantum noise (measurement back-action)
      float noise = random(vUv + time) * 0.02 * measurementStrength;
      diagonal += noise;
      
      return vec4(diagonal, rho.a);
    }
    
    void main() {
      vec4 inputColor = texture2D(tDiffuse, vUv);
      
      // Apply Lindblad evolution
      vec4 evolved = lindbladEvolution(inputColor, time * 0.016);
      
      // Measurement collapse effect (gradual)
      float collapse = smoothstep(0.0, 1.0, measurementStrength);
      
      // Snap to classical values when measured
      vec3 classical = floor(evolved.rgb * 4.0) / 4.0;
      vec3 quantum = evolved.rgb;
      
      vec3 finalColor = mix(quantum, classical, collapse);
      
      // Add decoherence shimmer
      float shimmer = sin(vUv.x * 100.0 + time * 10.0) * 0.02 * (1.0 - collapse);
      finalColor += shimmer;
      
      gl_FragColor = vec4(finalColor, 1.0);
    }
  `
};

// QUANTUM INTERFERENCE PASS - Young's Double Slit
// ====================================================

export const InterferenceShader = {
  uniforms: {
    tDiffuse: { value: null },
    time: { value: 0 },
    wavelength: { value: 0.05 }, // Normalized wavelength
    slitSeparation: { value: 0.1 },
    coherenceLength: { value: 0.5 },
    resolution: { value: new THREE.Vector2(1, 1) }
  },
  
  vertexShader: `
    varying vec2 vUv;
    void main() {
      vUv = uv;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  
  fragmentShader: `
    uniform sampler2D tDiffuse;
    uniform float time;
    uniform float wavelength;
    uniform float slitSeparation;
    uniform float coherenceLength;
    uniform vec2 resolution;
    
    varying vec2 vUv;
    
    #define M_PI 3.14159265359
    
    // Young's double slit interference pattern
    float interferencePattern(vec2 position) {
      float screenDistance = 1.0;
      
      // Path difference from two slits
      float pathDifference = slitSeparation * position.x / screenDistance;
      float phaseDifference = 2.0 * M_PI * pathDifference / wavelength;
      
      // Intensity: I = 4I₀cos²(Δφ/2)
      float intensity = cos(phaseDifference * 0.5);
      intensity = intensity * intensity;
      
      // Envelope (diffraction from single slit)
      float slitWidth = wavelength * 2.0;
      float sinc_arg = M_PI * slitWidth * position.x / (wavelength * screenDistance);
      float envelope = 1.0;
      if (abs(sinc_arg) > 0.01) {
        envelope = sin(sinc_arg) / sinc_arg;
        envelope = envelope * envelope;
      }
      
      return intensity * envelope;
    }
    
    // Coherence degradation with path length
    float coherenceFactor(float pathDifference) {
      return exp(-abs(pathDifference) / coherenceLength);
    }
    
    void main() {
      vec4 color = texture2D(tDiffuse, vUv);
      
      // Map UV to screen position
      vec2 position = vUv * 2.0 - 1.0;
      
      // Compute interference
      float interference = interferencePattern(position);
      float coherence = coherenceFactor(position.x);
      
      // Blend interference with scene
      float effectStrength = 0.15;
      float pattern = mix(1.0, interference, effectStrength * coherence);
      
      // Apply pattern to color
      vec3 finalColor = color.rgb * pattern;
      
      // Add subtle interference fringes
      float fringes = sin(position.x * 50.0 - time) * 0.03 * coherence;
      finalColor += fringes;
      
      gl_FragColor = vec4(finalColor, color.a);
    }
  `
};

// UNCERTAINTY PRINCIPLE PASS - Heisenberg Blur
// ====================================================

export const UncertaintyShader = {
  uniforms: {
    tDiffuse: { value: null },
    time: { value: 0 },
    positionUncertainty: { value: 0.002 },
    momentumUncertainty: { value: 100.0 },
    hbar: { value: 1.0 },
    resolution: { value: new THREE.Vector2(1, 1) }
  },
  
  vertexShader: `
    varying vec2 vUv;
    void main() {
      vUv = uv;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  
  fragmentShader: `
    uniform sampler2D tDiffuse;
    uniform float time;
    uniform float positionUncertainty;
    uniform float momentumUncertainty;
    uniform float hbar;
    uniform vec2 resolution;
    
    varying vec2 vUv;
    
    // Pseudo-random
    float random(vec2 st) {
      return fract(sin(dot(st.xy, vec2(12.9898, 78.233))) * 43758.5453123);
    }
    
    vec2 randomVec2(vec2 st) {
      return vec2(
        random(st),
        random(st + vec2(0.5, 0.5))
      ) * 2.0 - 1.0;
    }
    
    void main() {
      // Heisenberg limit: ΔxΔp ≥ ħ/2
      float uncertaintyProduct = positionUncertainty * momentumUncertainty;
      float heisenbergLimit = hbar / 2.0;
      
      // How much we violate/satisfy the limit
      float heisenbergFactor = uncertaintyProduct / heisenbergLimit;
      
      // Position uncertainty creates blur
      vec2 offset = randomVec2(vUv + time) * positionUncertainty * heisenbergFactor;
      
      // Sample with uncertainty
      vec4 color = texture2D(tDiffuse, vUv + offset);
      
      // Momentum uncertainty creates chromatic aberration
      float chromaticShift = 0.001 * heisenbergFactor;
      float r = texture2D(tDiffuse, vUv + offset + vec2(chromaticShift, 0.0)).r;
      float g = texture2D(tDiffuse, vUv + offset).g;
      float b = texture2D(tDiffuse, vUv + offset - vec2(chromaticShift, 0.0)).b;
      
      color.rgb = vec3(r, g, b);
      
      // Add quantum noise proportional to uncertainty
      float noise = random(vUv + time) * 0.02 * min(heisenbergFactor, 1.0);
      color.rgb += noise;
      
      gl_FragColor = color;
    }
  `
};

export const QuantumFoamShader = {
  uniforms: {
    tDiffuse: { value: null },
    time: { value: 0 },
    foamIntensity: { value: 0.5 },
    planckScale: { value: 0.001 },
    resolution: { value: new THREE.Vector2(1, 1) }
  },
  
  vertexShader: `
    varying vec2 vUv;
    void main() {
      vUv = uv;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  
  fragmentShader: `
    uniform sampler2D tDiffuse;
    uniform float time;
    uniform float foamIntensity;
    uniform float planckScale;
    uniform vec2 resolution;
    
    varying vec2 vUv;
    
    // 2D noise function
    float noise(vec2 p) {
      vec2 i = floor(p);
      vec2 f = fract(p);
      f = f * f * (3.0 - 2.0 * f);
      
      float a = fract(sin(dot(i, vec2(127.1, 311.7))) * 43758.5453);
      float b = fract(sin(dot(i + vec2(1.0, 0.0), vec2(127.1, 311.7))) * 43758.5453);
      float c = fract(sin(dot(i + vec2(0.0, 1.0), vec2(127.1, 311.7))) * 43758.5453);
      float d = fract(sin(dot(i + vec2(1.0, 1.0), vec2(127.1, 311.7))) * 43758.5453);
      
      return mix(mix(a, b, f.x), mix(c, d, f.x), f.y);
    }
    
    // Fractal brownian motion for foam texture
    float fbm(vec2 p) {
      float value = 0.0;
      float amplitude = 0.5;
      float frequency = 1.0;
      
      for (int i = 0; i < 5; i++) {
        value += amplitude * noise(p * frequency);
        frequency *= 2.0;
        amplitude *= 0.5;
      }
      
      return value;
    }
    
    void main() {
      vec2 pixelSize = 1.0 / resolution;
      
      // Quantum foam distortion
      float foamScale = 100.0 / planckScale;
      vec2 foamPos = vUv * foamScale + time * 0.5;
      
      float foam = fbm(foamPos) * foamIntensity * planckScale;
      
      // Virtual particle creation/annihilation
      float particles = step(0.98, noise(foamPos * 10.0 + time * 5.0)) * 0.1;
      
      // Apply foam distortion
      vec2 distortedUv = vUv + vec2(foam, foam * 0.7);
      
      vec4 color = texture2D(tDiffuse, distortedUv);
      
      // Add virtual particle flashes
      color.rgb += particles * vec3(0.3, 0.5, 1.0);
      
      // Metric fluctuation effect (subtle distortion)
      float metricFlux = sin(vUv.x * 200.0 + time * 3.0) * 
                         sin(vUv.y * 200.0 - time * 2.0) * 0.005 * foamIntensity;
      color.rgb += metricFlux;
      
      gl_FragColor = color;
    }
  `
};

export const EntanglementVisualizationShader = {
  uniforms: {
    tDiffuse: { value: null },
    time: { value: 0 },
    entanglementMap: { value: null },
    bellViolation: { value: 2.0 },
    resolution: { value: new THREE.Vector2(1, 1) }
  },
  
  vertexShader: `
    varying vec2 vUv;
    void main() {
      vUv = uv;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  
  fragmentShader: `
    uniform sampler2D tDiffuse;
    uniform float time;
    uniform sampler2D entanglementMap;
    uniform float bellViolation;
    uniform vec2 resolution;
    
    varying vec2 vUv;
    
    #define TSIRELSON_BOUND 2.828427125 // 2√2
    
    vec3 hsv2rgb(vec3 c) {
      vec4 K = vec4(1.0, 2.0 / 3.0, 1.0 / 3.0, 3.0);
      vec3 p = abs(fract(c.xxx + K.xyz) * 6.0 - K.www);
      return c.z * mix(K.xxx, clamp(p - K.xxx, 0.0, 1.0), c.y);
    }
    
    void main() {
      vec4 color = texture2D(tDiffuse, vUv);
      
      // Check entanglement strength at this location
      float entanglement = 0.0;
      if (entanglementMap != null) {
        entanglement = texture2D(entanglementMap, vUv).r;
      }
      
      // Bell inequality visualization
      // Classical limit: 2.0, Quantum maximum: 2√2
      float quantumness = smoothstep(2.0, TSIRELSON_BOUND, bellViolation);
      
      // Entanglement color (purple for quantum correlations)
      vec3 entanglementColor = hsv2rgb(vec3(0.8, 1.0, quantumness));
      
      // Add entanglement glow
      float glow = sin(vUv.x * 50.0 - time * 2.0) * sin(vUv.y * 50.0 + time * 1.5);
      glow = pow(abs(glow), 3.0) * quantumness * 0.2;
      
      color.rgb = mix(color.rgb, entanglementColor, glow);
      
      // Non-local correlation lines
      float correlationLine = step(0.99, sin(vUv.x * 100.0 + vUv.y * 100.0 + time * 5.0));
      color.rgb += correlationLine * entanglementColor * 0.1 * quantumness;
      
      gl_FragColor = color;
    }
  `
};

// QUANTUM POST-PROCESSING COMPOSER

export class QuantumPostProcessing {
  public composer: EffectComposer;
  private renderer: THREE.WebGLRenderer;
  private scene: THREE.Scene;
  private camera: THREE.Camera;
  
  private decoherencePass: ShaderPass;
  private interferencePass: ShaderPass;
  private uncertaintyPass: ShaderPass;
  private quantumFoamPass: ShaderPass;
  private entanglementPass: ShaderPass;
  private bloomPass: UnrealBloomPass;
  
  // Effect parameters
  public decoherenceRate: number = 0.1;
  public measurementStrength: number = 0.0;
  public coherenceLength: number = 0.5;
  public positionUncertainty: number = 0.002;
  public momentumUncertainty: number = 100;
  public foamIntensity: number = 0.3;
  public bellViolation: number = 2.5;
  
  // Enable/disable effects
  public enableDecoherence: boolean = true;
  public enableInterference: boolean = true;
  public enableUncertainty: boolean = true;
  public enableQuantumFoam: boolean = true;
  public enableEntanglement: boolean = true;
  public enableBloom: boolean = true;

  constructor(
    renderer: THREE.WebGLRenderer, 
    scene: THREE.Scene, 
    camera: THREE.Camera
  ) {
    this.renderer = renderer;
    this.scene = scene;
    this.camera = camera;
    
    // Create composer
    this.composer = new EffectComposer(renderer);
    
    // Base render pass
    const renderPass = new RenderPass(scene, camera);
    this.composer.addPass(renderPass);
    
    // Quantum effect passes
    this.decoherencePass = new ShaderPass(DecoherenceShader);
    this.interferencePass = new ShaderPass(InterferenceShader);
    this.uncertaintyPass = new ShaderPass(UncertaintyShader);
    this.quantumFoamPass = new ShaderPass(QuantumFoamShader);
    this.entanglementPass = new ShaderPass(EntanglementVisualizationShader);
    
    // Bloom for glow effects
    const size = renderer.getSize(new THREE.Vector2());
    this.bloomPass = new UnrealBloomPass(
      size,
      0.5,  // strength
      0.4,  // radius
      0.85  // threshold
    );
    
    // Add passes in order
    this.composer.addPass(this.decoherencePass);
    this.composer.addPass(this.interferencePass);
    this.composer.addPass(this.uncertaintyPass);
    this.composer.addPass(this.quantumFoamPass);
    this.composer.addPass(this.entanglementPass);
    this.composer.addPass(this.bloomPass);
    
    // Initial update
    this.updatePassStates();
  }

  updatePassStates(): void {
    this.decoherencePass.enabled = this.enableDecoherence;
    this.interferencePass.enabled = this.enableInterference;
    this.uncertaintyPass.enabled = this.enableUncertainty;
    this.quantumFoamPass.enabled = this.enableQuantumFoam;
    this.entanglementPass.enabled = this.enableEntanglement;
    this.bloomPass.enabled = this.enableBloom;
  }

  update(time: number): void {
    // Update all shader uniforms
    
    // Decoherence
    this.decoherencePass.uniforms.time.value = time;
    this.decoherencePass.uniforms.decoherenceRate.value = this.decoherenceRate;
    this.decoherencePass.uniforms.measurementStrength.value = this.measurementStrength;
    
    // Interference
    this.interferencePass.uniforms.time.value = time;
    this.interferencePass.uniforms.coherenceLength.value = this.coherenceLength;
    
    // Uncertainty (Heisenberg)
    this.uncertaintyPass.uniforms.time.value = time;
    this.uncertaintyPass.uniforms.positionUncertainty.value = this.positionUncertainty;
    this.uncertaintyPass.uniforms.momentumUncertainty.value = this.momentumUncertainty;
    this.uncertaintyPass.uniforms.hbar.value = 1.0; // Normalized
    
    // Quantum foam
    this.quantumFoamPass.uniforms.time.value = time;
    this.quantumFoamPass.uniforms.foamIntensity.value = this.foamIntensity;
    
    // Entanglement
    this.entanglementPass.uniforms.time.value = time;
    this.entanglementPass.uniforms.bellViolation.value = this.bellViolation;
    
    // Update pass states
    this.updatePassStates();
  }

  render(): void {
    this.composer.render();
  }

  resize(width: number, height: number): void {
    this.composer.setSize(width, height);
    
    const resolution = new THREE.Vector2(width, height);
    
    this.decoherencePass.uniforms.resolution.value = resolution;
    this.interferencePass.uniforms.resolution.value = resolution;
    this.uncertaintyPass.uniforms.resolution.value = resolution;
    this.quantumFoamPass.uniforms.resolution.value = resolution;
    this.entanglementPass.uniforms.resolution.value = resolution;
  }

  // Trigger measurement (wave function collapse)
  triggerMeasurement(duration: number = 0.5): void {
    const startTime = Date.now();
    
    const animate = () => {
      const elapsed = (Date.now() - startTime) / 1000;
      
      if (elapsed < duration) {
        // Ramp up measurement strength
        this.measurementStrength = Math.sin(elapsed / duration * Math.PI);
        requestAnimationFrame(animate);
      } else {
        // Collapse complete
        this.measurementStrength = 0;
      }
    };
    
    animate();
  }

  // Set quantum state coherence
  setCoherence(level: number): void {
    // level: 0 = fully decohered, 1 = fully coherent
    this.decoherenceRate = (1 - level) * 0.5;
    this.coherenceLength = level;
    this.positionUncertainty = (1 - level) * 0.01 + 0.001;
  }

  // Set Bell inequality violation level
  setBellViolation(level: number): void {
    // level: 0-1 maps to 2.0 - 2√2
    this.bellViolation = 2.0 + level * 0.828427125;
  }

  dispose(): void {
    this.composer.dispose();
  }
}

export default QuantumPostProcessing;
