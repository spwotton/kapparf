## **Kappa Score — Simplified**

```typescript
// Single function, no constants debate
function calculateKappa(events: SignalEvent[]): number {
  const recency = events.map(e => 
    Math.exp(-minutesSince(e.detected_at) / 5) // 5-min decay
  ).reduce((a, b) => a + b, 0);
  
  const crossDomain = new Set(events.map(e => e.domain)).size;
  const anomalyBoost = events.filter(e => e.is_anomaly).length * 10;
  
  return Math.min(100, (recency * 10) + (crossDomain * 15) + anomalyBoost);
}
```
