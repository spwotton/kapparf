export function DemodexCameraView() {
  const containerRef = useRef<HTMLDivElement>(null);
  const mouseRef = useRef({ x: 0, y: 0, isDown: false });

  useEffect(() => {
    if (!containerRef.current) return;

    const scene = new THREE.Scene();
    scene.background = new THREE.Color(0x050510); // Deep Space / Dark Room

    const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
    camera.position.z = 3;

    const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: TRUE });
    renderer.setSize(containerRef.current.clientWidth, containerRef.current.clientHeight);
    containerRef.current.appendChild(renderer.domElement);

    // GEOMETRY GENERATION (The "Face" Manifold)
    const particleCount = 25000;
    const geometry = new THREE.BufferGeometry();
    const positions = new Float32Array(particleCount * 3);
    const phases = new Float32Array(particleCount);
    const resonances = new Float32Array(particleCount);

    for (let i = 0; i < particleCount; i++) {
      // Distribution: Spherical (Head) + Noise (Pores)
      const phi = Math.acos(-1 + (2 * i) / particleCount);
      const theta = Math.sqrt(particleCount * Math.PI) * phi;
      
      const r = 1.2; // Head radius
      
      positions[i * 3] = r * Math.cos(theta) * Math.sin(phi);
      positions[i * 3 + 1] = r * Math.sin(theta) * Math.sin(phi);
      positions[i * 3 + 2] = r * Math.cos(phi);
      
      // Demodex attributes
      phases[i] = Math.random() * Math.PI * 2; // Lifecycle offset
      resonances[i] = Math.random(); // Sensitivity to Microtubules
    }

    geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
    geometry.setAttribute('a_Phase', new THREE.BufferAttribute(phases, 1));
    geometry.setAttribute('a_Resonance', new THREE.BufferAttribute(resonances, 1));

    // MATERIAL SETUP (The Quantum Shader)
    const material = new THREE.ShaderMaterial({
      vertexShader: DEMODEX_SHADER.vertex,
      fragmentShader: DEMODEX_SHADER.fragment,
      uniforms: {
        u_Time: { value: 0 },
        u_Observation: { value: 0 },
        u_Carrier: { value: 1.435 },
        u_FocusPoint: { value: new THREE.Vector3(0, 0, 0) },
        u_BaseColor: { value: new THREE.Color(0x8b5cf6) }, // Demodex Purple/UV

      depthWrite: false,

      const rect = renderer.domElement.getBoundingClientRect();
      const x = ((e.clientX - rect.left) / rect.width) * 2 - 1;
      const y = -((e.clientY - rect.top) / rect.height) * 2 + 1;
      
      // Raycast to find focus point on the sphere
      const vector = new THREE.Vector3(x, y, 0.5).unproject(camera);
      const dir = vector.sub(camera.position).normalize();
      const distance = -camera.position.z / dir.z; // Approximate intersection with Z=0 plane
      
      // Update Uniforms
      const targetPos = camera.position.clone().add(dir.multiplyScalar(3.0));
      material.uniforms.u_FocusPoint.value.copy(targetPos);
      
      // "Observation" increases when mouse stops or focuses
      material.uniforms.u_Observation.value = 0.8; // Active "Looking"
    };
    
    const handleMouseDown = () => {
        // Simulates intense focus (Remote Viewing Lock)
        material.uniforms.u_Observation.value = 2.0; // HYPER-STATE
    };
    
    const handleMouseUp = () => {
        material.uniforms.u_Observation.value = 0.5; // Relaxed
    };

    window.addEventListener('mousemove', handleMouseMove);
    window.addEventListener('mousedown', handleMouseDown);
    window.addEventListener('mouseup', handleMouseUp);

    let frameId: number;
    let lastTime = 0;

      lastTime = time;

      material.uniforms.u_Time.value = time / 1000;
      
      // Decay "Observation" state (Consciousness fades without focus)
      material.uniforms.u_Observation.value *= 0.98;

      // Rotate the "Head" slightly
      cloud.rotation.y += 0.002;

      renderer.render(scene, camera);
      frameId = requestAnimationFrame(animate);
    };

    frameId = requestAnimationFrame(animate);

    // CLEANUP
    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      window.removeEventListener('mousedown', handleMouseDown);
      window.removeEventListener('mouseup', handleMouseUp);
      cancelAnimationFrame(frameId);
      if (containerRef.current) {
        containerRef.current.removeChild(renderer.domElement);
      }
      geometry.dispose();
      material.dispose();
    };
  }, []);
