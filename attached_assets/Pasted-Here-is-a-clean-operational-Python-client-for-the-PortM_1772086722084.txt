class PortMasterNexus:

        self.timeout = timeout

        """Verify API connectivity"""
        try:
            r = self.session.get(f"{self.base_url}/api/core/firmware/status",

            return r.status_code == 200
        except:
            return False

    def _request(self, method: str, endpoint: str, data: Optional[Dict] = None) -> Dict:
        """Internal request handler with error wrapping"""
        url = f"{self.base_url}{endpoint}"
        try:
            if method == "GET":
                r = self.session.get(url, timeout=self.timeout)
            elif method == "POST":
                r = self.session.post(url, json=data, timeout=self.timeout)
            elif method == "DELETE":
                r = self.session.delete(url, timeout=self.timeout)
            else:
                raise ValueError(f"Unsupported method: {method}")
            
            r.raise_for_status()
            return r.json()
            
        except requests.exceptions.HTTPError as e:
            if r.status_code == 401:
                raise PermissionError("🔑 API Key rejected - check 87453e0d-f4e3...")
            elif r.status_code == 404:
                raise FileNotFoundError(f"Endpoint {endpoint} not found")
            else:
                raise RuntimeError(f"API Error {r.status_code}: {e}")
        except requests.exceptions.ConnectionError:
            raise ConnectionError(f"💀 PortMaster unreachable at {self.base_url}")

            print(f"✅ Rule {rule_id} removed")

        except Exception as e:

    def download_capture(self, capture_id: str, output_file: str = None) -> bytes:
        """Download PCAP data for analysis in Wireshark"""

            for chunk in r.iter_content(chunk_size=8192):
                f.write(chunk)
        
        print(f"💾 Saved to {output_file}")
        return output_file
