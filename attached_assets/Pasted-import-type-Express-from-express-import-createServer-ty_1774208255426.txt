import type { Express } from "express";
import { createServer, type Server } from "http";

import fs from "fs";
import path from "path";
import { pipeline } from "./pipeline/pipeline";

const serverStartTime = Date.now();

export async function registerRoutes(
  httpServer: Server,
  app: Express
): Promise<Server> {

  app.get("/api/thoughts", async (_req, res) => {
    const thoughts = await storage.getThoughts();
    res.json(thoughts);
  });

  app.get("/api/papers", async (_req, res) => {
    const papers = await storage.getPapers();
    res.json(papers);
  });

  app.post("/api/papers", async (req, res) => {
    try {
      const data = insertPaperSchema.parse(req.body);
      const paper = await storage.createPaper(data);
      res.status(201).json(paper);
    } catch (e) {
      if (e instanceof ZodError) {
        res.status(400).json({ error: e.errors });
      } else {
        res.status(500).json({ error: "Failed to create paper" });
      }
    }
  });

  app.get("/api/constants", async (_req, res) => {
    const constants = await storage.getConstants();
    res.json(constants);
  });

  app.get("/api/resonome", async (_req, res) => {
    const genes = await storage.getResonomeGenes();
    res.json(genes);
  });

  app.get("/api/evidence-audit", async (_req, res) => {
    const audit = await storage.getEvidenceAudit();
    res.json(audit);
  });

  app.get("/api/engine/frames", async (req, res) => {
    try {
      const limit = req.query.limit ? parseInt(req.query.limit as string, 10) : 50;
      const frames = await storage.getEngineFrames(limit);
      res.json(frames);
    } catch (e) {
      res.status(500).json({ error: "Failed to fetch engine frames" });
    }
  });

  app.get("/api/engine/latest", async (_req, res) => {
    try {
      const frame = await storage.getLatestEngineFrame();
      if (!frame) {
        res.status(404).json({ error: "No engine frames available" });
        return;
      }
      res.json(frame);
    } catch (e) {
      res.status(500).json({ error: "Failed to fetch latest engine frame" });
    }
  });

  app.get("/api/engine/quantum", async (_req, res) => {
    try {
      const frame = await storage.getLatestEngineFrame();
      if (!frame) {
        res.json({
          coherence: 0,
          psiUnity: 0,
          quantumSignature: "",
          frameSignature: "",
          pllLock: 0,
          pllJitter: 0,
          quboEnergy: 0,
          wukongBandwidth: 0,
          wukongQuality: 0,
          activeSNPs: [],
          timestamp: null,
        });
        return;
      }
      
      let fullFrameData: any = {};
      try {
        fullFrameData = JSON.parse(frame.fullFrame);
      } catch {
        // If parsing fails, return defaults
      }
      
      const signal = fullFrameData.signal ?? {};
      
      res.json({
        coherence: signal.quantum_coherence ?? 0,
        psiUnity: signal.quantum_psi ?? 0,
        quantumSignature: signal.quantum_signature ?? "",
        frameSignature: signal.quantum_signature ?? "",
        pllLock: signal.pll_lock ?? 0,
        pllJitter: signal.pll_jitter ?? 0,
        quboEnergy: signal.qubo_energy ?? 0,
        wukongBandwidth: signal.wukong_bandwidth ?? 0,
        wukongQuality: signal.wukong_quality ?? 0,
        activeSNPs: fullFrameData.active_snps ?? [],
        timestamp: frame.createdAt,
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to fetch quantum metrics" });
    }
  });

  app.get("/api/engine/doppler", async (_req, res) => {
    try {
      const frame = await storage.getLatestEngineFrame();
      if (!frame) {
        res.json({
          frftPeak: 0,
          frftAlpha: 0,
          dopplerCoherence: 0,
          resolution: 0,
          instantFreqMean: 0,
          timestamp: null,
        });
        return;
      }
      
      let fullFrameData: any = {};
      try {
        fullFrameData = JSON.parse(frame.fullFrame);
      } catch {
        // If parsing fails, return defaults
      }
      
      const signal = fullFrameData.signal ?? {};
      
      res.json({
        frftPeak: signal.frft_peak ?? 0,
        frftAlpha: signal.frft_alpha ?? 0,
        dopplerCoherence: signal.doppler_coherence ?? 0,
        resolution: signal.doppler_resolution ?? 0,
        instantFreqMean: signal.doppler_instant_freq_mean ?? 0,
        timestamp: frame.createdAt,
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to fetch Doppler metrics" });
    }
  });

  app.get("/api/engine/pathways", async (_req, res) => {
    try {
      const frame = await storage.getLatestEngineFrame();
      if (!frame) {
        res.json({
          activePathways: {},
          healthStatus: "OK",
          conharmonic: { cos2theta: 0, invPhi2: 0, epsilon: 0, isValid: false },
          welch: { dominantFreq: 0, spectralCentroid: 0, spectralFlatness: 0 },
          timestamp: null,
        });
        return;
      }

      let fullFrameData: any = {};
      try {
        fullFrameData = JSON.parse(frame.fullFrame);
      } catch {}

      const signal = fullFrameData.signal ?? {};
      const frftPeak = signal.frft_peak ?? 0;

      const activePathwaysMap = quantumResonome.pathways.detectActivePathways([frftPeak], 5.0);
      const activePathways: Record<string, number[]> = {};
      activePathwaysMap.forEach((freqs, category) => {
        activePathways[category] = freqs;
      });

      const healthStatus = quantumResonome.pathways.deriveHealthStatus(new Set(Object.keys(activePathways)));
      const conharmonic = quantumResonome.conharmonic.validate();

      res.json({
        activePathways,
        healthStatus,
        conharmonic,
        welch: {
          dominantFreq: signal.welch_dominant_freq ?? 0,
          spectralCentroid: signal.welch_spectral_centroid ?? 0,
          spectralFlatness: signal.welch_spectral_flatness ?? 0,
        },
        timestamp: frame.createdAt,
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to fetch pathway data" });
    }
  });

  app.get("/api/engine/kuramoto", async (_req, res) => {
    try {
      const frame = await storage.getLatestEngineFrame();
      if (!frame) {
        res.json({
          orderParameter: 0,
          meanPhase: 0,
          conharmonicDrift: 0,
          timestamp: null,
        });
        return;
      }

      let fullFrameData: any = {};
      try {
        fullFrameData = JSON.parse(frame.fullFrame);
      } catch {}

      const signal = fullFrameData.signal ?? {};

      const phases: number[] = [];
      if (fullFrameData.token && fullFrameData.kalman_confidence !== undefined) {
        const projectedKeys = [
          signal.frft_peak ?? 0,
          signal.pll_lock ?? 0,
          signal.doppler_coherence ?? 0,
          signal.wukong_bandwidth ?? 0,
          signal.wukong_quality ?? 0,
          signal.qubo_energy ?? 0,
          signal.quantum_coherence ?? 0,
          signal.quantum_psi ?? 0,
        ];
        for (const v of projectedKeys) {
          phases.push(v);
        }
      }

      if (phases.length === 0) {
        phases.push(0, 0, 0, 0, 0, 0, 0, 0);
      }

      const kuramoto = quantumResonome.kuramoto.orderParameter(phases);
      const conharmonicDrift = quantumResonome.conharmonic.driftMonitor(phases);

      res.json({
        orderParameter: kuramoto.r,
        meanPhase: kuramoto.meanPhase,
        conharmonicDrift,
        timestamp: frame.createdAt,
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to fetch Kuramoto data" });
    }
  });

  app.get("/api/engine/status", async (_req, res) => {
    try {
      const maxFrame = await storage.getMaxFrameNumber();
      const latestFrame = await storage.getLatestEngineFrame();
      const uptimeMs = Date.now() - serverStartTime;

      res.json({
        connected: maxFrame > 0,
        frameCount: maxFrame,
        latestTimestamp: latestFrame?.createdAt ?? null,
        latestFrameNumber: latestFrame?.frameNumber ?? null,
        uptimeSeconds: Math.floor(uptimeMs / 1000),
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to fetch engine status" });
    }
  });

  app.get("/api/engine/webhook", (_req, res) => {
    const filePath = path.resolve(process.cwd(), "public/kyma_webhook.py");
    if (fs.existsSync(filePath)) {
      res.setHeader("Content-Type", "text/plain");
      res.setHeader("Content-Disposition", "attachment; filename=kyma_webhook.py");
      fs.createReadStream(filePath).pipe(res);
    } else {
      res.status(404).json({ error: "Webhook file not found" });
    }
  });

  app.post("/api/engine/frames", async (req, res) => {
    try {
      const body = req.body;

      const tokenLabel = body.token?.label ?? "";
      const labelParts = tokenLabel.split("_");
      const dominantState = (labelParts.length > 0 && labelParts[0]) ? labelParts[0] : "UNKNOWN";

      const mappedData = {
        frameNumber: body.frame,
        decodedText: body.decoded?.fused_text ?? "",
        dominantState,
        behavior: body.decoded?.behavior ?? "",
        intentionText: body.decoded?.intention_text ?? null,
        flowRegime: body.flow?.regime ?? "unknown",
        flowConfidence: body.flow?.viscous_dominance ?? null,
        kalmanConfidence: body.kalman_confidence ?? null,
        psiT: body.psi_t ?? null,
        rotationTilt: body.rotation?.tilt ?? null,
        rotationAlignment: body.rotation?.alignment ?? null,
        bellS: body.rotation?.bell_S ?? null,
        apertureLocked: body.rotation?.aperture_locked ?? false,
        tokenLabel: body.token?.label ?? null,
        tokenConfidence: body.token?.conf ?? null,
        sensorStatus: body.sensors ? JSON.stringify(body.sensors) : null,
        fullFrame: JSON.stringify(body),
      };

      const validated = insertEngineFrameSchema.parse(mappedData);
      const frame = await storage.createEngineFrame(validated);

      const frameCount = await storage.getEngineFrameCount();
      if (frameCount >= 5 && (frameCount <= 10 || frameCount % 10 === 0)) {
        trainCodebook().catch(() => {});
      }

      if (mappedData.decodedText) {
        tokenizeFrame(mappedData.decodedText, mappedData.dominantState).catch(() => {});
      }

      res.status(201).json(frame);
    } catch (e) {
      if (e instanceof ZodError) {
        res.status(400).json({ error: e.errors });
      } else {
        res.status(500).json({ error: "Failed to create engine frame" });
      }
    }
  });

  app.get("/api/ml/status", async (_req, res) => {
    try {
      const status = await getMLStatus();
      res.json(status);
    } catch (e) {
      res.status(500).json({ error: "Failed to fetch ML status" });
    }
  });

  app.post("/api/ml/train", async (_req, res) => {
    try {
      const result = await trainCodebook();
      res.json(result);
    } catch (e) {
      console.error("ML training error:", e);
      res.status(500).json({ error: "Training failed" });
    }
  });

  app.get("/api/ml/codebook", async (_req, res) => {
    try {
      const codebook = await storage.getCodebook();
      res.json(codebook);
    } catch (e) {
      res.status(500).json({ error: "Failed to fetch codebook" });
    }
  });

  app.get("/api/ml/quantize", async (_req, res) => {
    try {
      const frame = await storage.getLatestEngineFrame();
      if (!frame) {
        res.status(404).json({ error: "No frames available" });
        return;
      }
      const result = await quantizeFrame(frame);
      if (!result) {
        res.status(404).json({ error: "Codebook not trained yet" });
        return;
      }
      res.json(result);
    } catch (e) {
      res.status(500).json({ error: "Failed to quantize frame" });
    }
  });

  app.get("/api/rsnc/stats", async (_req, res) => {
    try {
      const stats = stratifiedCodex.getStats();
      res.json(stats);
    } catch (e) {
      res.status(500).json({ error: "Failed to get RSNC stats" });
    }

  app.get("/api/rsnc/archetypes", async (_req, res) => {
    try {
      res.json(ARCHETYPE_DEFINITIONS);
    } catch (e) {
      res.status(500).json({ error: "Failed to get archetypes" });
    }
  });

  app.get("/api/rsnc/v4/status", async (_req, res) => {
    try {
      const stats = stratifiedCodex.getStats();
      const compounds = await storage.getPhytoCompounds();
      const protocols = await storage.getArchetypeProtocols();
      res.json({
        ...stats,
        phytoCompoundCount: compounds.length,
        protocolCount: protocols.length,
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to get RSNC v4 status" });
    }
  });

  app.get("/api/rsnc/phyto-compounds", async (_req, res) => {
    try {
      const compounds = await storage.getPhytoCompounds();
      const geneEffects = await storage.getCompoundGeneEffects();
      const enriched = compounds.map(c => ({
        ...c,
        geneEffects: geneEffects.filter(ge => ge.compoundId === c.compoundId),
      }));
      res.json(enriched);
    } catch (e) {
      res.status(500).json({ error: "Failed to get phyto compounds" });
    }
  });

  app.get("/api/rsnc/phyto-protocols", async (_req, res) => {
    try {
      const archetypeIndex = _req.query.archetype ? parseInt(_req.query.archetype as string) : undefined;
      const protocols = await storage.getArchetypeProtocols(archetypeIndex);
      res.json(protocols);
    } catch (e) {
      res.status(500).json({ error: "Failed to get phyto protocols" });
    }
  });

  app.get("/api/rsnc/data-sources", async (_req, res) => {
    try {
      const sources = await storage.getEegDataSources();
      res.json(sources);
    } catch (e) {
      res.status(500).json({ error: "Failed to get EEG data sources" });
    }
  });

  app.get("/api/rsnc/scaling-status", async (_req, res) => {
    try {
      const scalingStatus = stratifiedCodex.getScalingStatus();
      const sources = await storage.getEegDataSources();
      const { getQualityStats } = await import("./rsnc/eeg-data-registry");
      const qualityStats = getQualityStats(sources);
      res.json({ ...scalingStatus, dataRegistry: qualityStats });
    } catch (e) {
      res.status(500).json({ error: "Failed to get scaling status" });
    }
  });

  app.get("/api/rsnc/quality-stats", async (_req, res) => {
    try {
      const sources = await storage.getEegDataSources();
      const { getQualityStats, PREPROCESSING_PIPELINE: pp } = await import("./rsnc/eeg-data-registry");
      const qualityScores = await storage.getEpochQualityScores(100);
      const stats = getQualityStats(sources);
      const thresholds = {
        spectralFlatnessMax: pp.qualityThresholds.spectralFlatnessMax,
        lineNoiseMax: pp.qualityThresholds.lineNoiseMax,

        movementArtifactMax: pp.qualityThresholds.movementArtifactMax,
        overallRejectBelow: pp.qualityThresholds.overallRejectBelow,
      };
      res.json({ ...stats, recentScores: qualityScores, thresholds });
    } catch (e) {
      res.status(500).json({ error: "Failed to get quality stats" });
    }
  });

  app.get("/api/rsnc/phyto-recommendation", async (_req, res) => {
    try {
      const recommendation = stratifiedCodex.getPhytoRecommendation();
      const stats = stratifiedCodex.getStats();
      const archIdx = stats.gosLayer.index;
      const protocols = await storage.getArchetypeProtocols(stats.archetypeWeights.indexOf(Math.max(...stats.archetypeWeights)));
      res.json({
        recommendation,
        phoenixProgress: stats.phoenixProgress,
        gosLayer: stats.gosLayer,
        protocol: protocols[0] ?? null,
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to get phyto recommendation" });
    }
  });

  app.get("/api/rsnc/moonshine", async (_req, res) => {
    try {
      const validation = stratifiedCodex.validateMoonshine();
      const spokeMap = stratifiedCodex.getArchetypeSpokeMapping();
      const niemeierStats = stratifiedCodex.getNiemeierStats();
      res.json({
        moonshine: validation,
        archetypeSpokeMap: spokeMap,
        niemeier: niemeierStats,
        constants: {
          jInvariant: 744,
          monsterSmallestRep: 196883,
          monsterMultiplier: 4200,
          leechKissing: 196560,
          cell24Edges: 96,
          primeSpokes: [1, 5, 7, 11, 13, 17, 19, 23],
          cardinalFreqs: { A24: 37, D24: 46.875, E8_CUBED: 111, LEECH: 192 },
        },
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to get Moonshine data" });
    }
  });

  app.get("/api/rsnc/niemeier/classify", async (req, res) => {
    try {
      const freqStr = req.query.freq as string;
      if (!freqStr) {
        return res.status(400).json({ error: "freq parameter required" });
      }
      const freq = parseFloat(freqStr);
      if (isNaN(freq)) {
        return res.status(400).json({ error: "freq must be a number" });
      }
      const classification = stratifiedCodex.classifyFrequency(freq);
      res.json(classification);
    } catch (e) {
      res.status(500).json({ error: "Failed to classify frequency" });
    }
  });

  app.post("/api/rsnc/niemeier/classify-batch", async (req, res) => {
    try {
      const { frequencies } = req.body;
      if (!Array.isArray(frequencies)) {
        return res.status(400).json({ error: "frequencies array required" });
      }
      const result = stratifiedCodex.classifySignalFrequencies(frequencies);
      res.json(result);
    } catch (e) {
      res.status(500).json({ error: "Failed to classify frequencies" });
    }
  });

  app.get("/api/rsnc/wavelet/bands", async (_req, res) => {
    try {
      const { waveletBank } = await import("./rsnc/morlet-wavelet");
      const testSignal = new Float64Array(512);
      for (let i = 0; i < 512; i++) {
        testSignal[i] = Math.sin(2 * Math.PI * 10 * i / 256)
          + 0.5 * Math.sin(2 * Math.PI * 37 * i / 256)
          + 0.3 * Math.sin(2 * Math.PI * 6 * i / 256)
          + 0.2 * Math.sin(2 * Math.PI * 40 * i / 256);
      }
      const bands = waveletBank.extractBandPowers(testSignal);
      const pac = waveletBank.extractPAC(testSignal);
      const deathNote = waveletBank.checkDeathNote(testSignal);
      const features = waveletBank.extractResonomeFeatures(testSignal);
      res.json({
        bandPowers: bands,
        pac: {
          modulationIndex: pac.modulationIndex,
          tychoPower: pac.tychoPower,
          maggieSync: pac.maggieSync,
        },
        deathNote,
        resonomeFeatures: {
          thetaBetaRatio: features.thetaBetaRatio,
          alphaPower: features.alphaPower,
          alphaAsymmetry: features.alphaAsymmetry,
          muSuppression: features.muSuppression,
          pacGlobal: features.pacGlobal,
          pacLaterality: features.pacLaterality,
          spectralEntropy: features.spectralEntropy,
          thetaBurstRate: features.thetaBurstRate,
          tychoPower: features.tychoPower,
          maggieSync: features.maggieSync,
          deathNoteQuarantined: features.deathNoteQuarantined,
          epochQuality: features.epochQuality,
        },
        config: {
          sampleRate: 256,
          nCycles: 7,
          tychoCenterHz: 37,
          tychoBandwidthHz: 0.5,
          pacPhaseBins: 18,
        },
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to get wavelet band data" });
    }
  });

  app.get("/api/rsnc/melanin/state", async (req, res) => {
    try {
      const { MelaninTransducer } = await import("./rsnc/melanin-transducer");
      const tissueType = (req.query.tissue as string) || "EPIDERMIS";
      const validTypes = ["EPIDERMIS", "SUBSTANTIA_NIGRA", "RPE", "LOCUS_COERULEUS", "STRIA_VASCULARIS"];
      const tissue = validTypes.includes(tissueType) ? tissueType as any : "EPIDERMIS";

      const transducer = new MelaninTransducer(tissue);

      const hydration = req.query.hydration ? parseFloat(req.query.hydration as string) : undefined;
      const light = req.query.light ? parseFloat(req.query.light as string) : undefined;
      if (hydration !== undefined && isFinite(hydration)) transducer.setHydration(hydration);
      if (light !== undefined && isFinite(light)) transducer.setLightExposure(light);

      const metrics = transducer.getFullMetrics();
      res.json(metrics);
    } catch (e) {
      res.status(500).json({ error: "Failed to get melanin state" });
    }
  });

  app.get("/api/rsnc/melanin/piezo-sweep", async (req, res) => {
    try {
      const { MelaninTransducer } = await import("./rsnc/melanin-transducer");
      const transducer = new MelaninTransducer("EPIDERMIS");
      const start = Math.max(1, Math.min(500, parseFloat((req.query.start as string) || "50")));
      const end = Math.max(start + 1, Math.min(500, parseFloat((req.query.end as string) || "150")));
      const step = Math.max(0.1, parseFloat((req.query.step as string) || "0.5"));
      if (!isFinite(start) || !isFinite(end) || !isFinite(step)) {
        res.status(400).json({ error: "Invalid numeric parameters" });
        return;
      }
      const sweep = transducer.sweepPiezoFrequency(start, end, step);
      const resonantPoint = sweep.find(p => p.isResonant);
      res.json({
        sweep,
        resonantFrequency: 97.026,
        resonantPoint,
        falsifiable: "Peak in mechanical resonance and ionic-electronic charge transport at 97.026±0.1 Hz",
        protocol: "Melanin-sensitized QCM + swept acoustic excitation (50-150 Hz) + scanning LDV",
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to run piezo sweep" });
    }
  });

  app.get("/api/rsnc/melanin/hydration-sweep", async (req, res) => {
    try {
      const { MelaninTransducer } = await import("./rsnc/melanin-transducer");
      const transducer = new MelaninTransducer("EPIDERMIS");
      const steps = Math.max(2, Math.min(100, parseInt((req.query.steps as string) || "20")));
      const sweep = transducer.sweepHydration(steps);
      res.json({
        sweep,
        model: "Electronic-ionic hybrid conductor",
        mechanism: "Comproportionation equilibrium: hydration → semiquinone doping → protonic + electronic carriers",
        conductivityRange: { dry: 1e-8, hydrated: "1e-4 to 1e-3 S/cm" },
        muonHoppingRate: "0.28±0.03 μs⁻¹",
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to run hydration sweep" });
    }
  });

  app.get("/api/rsnc/melanin/bci-correction", async (req, res) => {
    try {
      const { MelaninTransducer } = await import("./rsnc/melanin-transducer");
      const transducer = new MelaninTransducer("EPIDERMIS");
      const basePower = Math.max(0.001, parseFloat((req.query.power as string) || "1.0"));
      const hydration = req.query.hydration ? parseFloat(req.query.hydration as string) : undefined;
      if (hydration !== undefined && isFinite(hydration)) transducer.setHydration(hydration);
      const correction = transducer.computeSNRCorrection(basePower);
      const metrics = transducer.getFullMetrics();
      res.json({
        correction,
        bciModel: metrics.bciCorrection,
        hypothesis: "SNR and decoding fidelity vary as a non-linear function of localized eumelanin ionic conductivity and semiquinone radical concentration",
        protocol: "DeWave-style discrete codex EEG + concurrent multi-harmonic EPR scanner",
        benchmarks: {

  app.get("/api/markets/portfolio", async (_req, res) => {
    try {
      const portfolio = await storage.getPaperPortfolio();
      const trades = await storage.getPaperTrades(portfolio.id);
      const positions: Record<string, { quantity: number; avgPrice: number }> = {};
      for (const trade of [...trades].reverse()) {
        if (!positions[trade.ticker]) positions[trade.ticker] = { quantity: 0, avgPrice: 0 };
        const pos = positions[trade.ticker];
        if (trade.action === "buy") {
          const totalCost = pos.avgPrice * pos.quantity + trade.price * trade.quantity;
          pos.quantity += trade.quantity;
          pos.avgPrice = pos.quantity > 0 ? totalCost / pos.quantity : 0;
        } else {
          pos.quantity -= trade.quantity;
          if (pos.quantity <= 0) { pos.quantity = 0; pos.avgPrice = 0; }
        }
      }
      const activePositions: { ticker: string; quantity: number; avgPrice: number; currentPrice?: number; pnl?: number }[] = [];
      let positionsValue = 0;
      for (const [ticker, pos] of Object.entries(positions)) {
        if (pos.quantity > 0) {
          const latest = await storage.getLatestMarketData(ticker);
          const currentPrice = latest?.close ?? pos.avgPrice;
          const pnl = (currentPrice - pos.avgPrice) * pos.quantity;
          positionsValue += currentPrice * pos.quantity;
          activePositions.push({ ticker, ...pos, currentPrice, pnl });
        }
      }
      res.json({
        ...portfolio,
        positions: activePositions,
        totalValue: portfolio.cash + positionsValue,
      });
    } catch (e) {
      res.status(500).json({ error: "Failed to get portfolio" });
    }
  });

  app.post("/api/markets/trade", async (req, res) => {
    try {
      const { ticker, action, quantity: rawQty } = req.body;
      if (!ticker || !action || !["buy", "sell"].includes(action)) {
        return res.status(400).json({ error: "ticker, action (buy/sell), quantity required" });
      }
      const quantity = Math.floor(Number(rawQty));
      if (!Number.isFinite(quantity) || quantity <= 0) {
        return res.status(400).json({ error: "Quantity must be a positive integer" });
      }
      const latest = await storage.getLatestMarketData(ticker);
      if (!latest) {
        return res.status(404).json({ error: `No market data for ${ticker}` });
      }
      const { generateCurrentSignals } = await import("./trading/backtest-engine");
      const recentData = await storage.getMarketData(ticker);
      const sortedData = recentData.sort((a, b) => a.date.localeCompare(b.date)).slice(-13);
      const allTickers = await storage.getMarketTickers();
      const tickerMeta = allTickers.find(t => t.ticker === ticker);
      const signal = generateCurrentSignals(sortedData, tickerMeta?.sector);
      const portfolio = await storage.getPaperPortfolio();
      const price = latest.close;
      const total = price * quantity;
      if (action === "buy" && total > portfolio.cash) {
        return res.status(400).json({ error: `Insufficient cash. Need $${total.toFixed(2)}, have $${portfolio.cash.toFixed(2)}` });
      }
      if (action === "sell") {
        const trades = await storage.getPaperTrades(portfolio.id);
        let held = 0;
        for (const t of trades) {
          if (t.ticker === ticker) held += t.action === "buy" ? t.quantity : -t.quantity;
        }
        if (quantity > held) {
          return res.status(400).json({ error: `Cannot sell ${quantity} shares of ${ticker}. You hold ${held}.` });
        }
      }
      const newCash = action === "buy" ? portfolio.cash - total : portfolio.cash + total;
      await storage.updatePortfolioCash(portfolio.id, newCash);
      const trade = await storage.createPaperTrade({
        portfolioId: portfolio.id,
        ticker,
        action,
        quantity,
        price,
        total,
        signal: signal.signal,
        signalStrength: signal.strength,
        kleinAngle: signal.kleinAngle,
        checksumValid: signal.checksumValid,
      });
      res.json({ trade, portfolio: { cash: newCash }, signal });
    } catch (e) {
      console.error("Trade error:", e);
      res.status(500).json({ error: "Trade failed" });
    }
  });

  app.post("/api/pipeline/process", async (_req, res) => {
    if (pipeline.isRunning()) {
      return res.status(409).json({ error: "Pipeline already running" });
    }
    pipeline.setRunning(true);
    res.json({ status: "started", message: "Pipeline processing started in background" });

    (async () => {
      try {
        const docs = await storage.getKnowledgeDocuments();
        let processed = 0;
        let skipped = 0;

        for (const doc of docs) {
          if (!doc.content || doc.content.length < 100) { skipped++; continue; }
          if (doc.content.startsWith("[PDF Reference]")) { skipped++; continue; }

          const existing = await storage.getPipelineArtifactByDocId(doc.id);
          if (existing) { skipped++; continue; }

          const maxChars = 200000;
          const content = doc.content.length > maxChars
            ? doc.content.substring(0, maxChars) + `\n\n[... truncated at ${maxChars} chars for pipeline processing ...]`
            : doc.content;

          try {
            const artifact = pipeline.processDocument(
              doc.id, doc.title, content,
              doc.source || "knowledge-base",
              doc.fileType || "text"
            );

            await storage.createPipelineArtifact({
              documentId: doc.id,
              title: doc.title,
              source: doc.source || "knowledge-base",
              bh53Id: artifact.stages.encoding.bh53Id,
              skeletonContent: artifact.stages.compression.skeleton.content.substring(0, 100000),
              digestContent: artifact.stages.compression.digest.content.substring(0, 100000),
              skeletonBytes: artifact.stages.compression.skeleton.outputBytes,
              digestBytes: artifact.stages.compression.digest.outputBytes,
              rawBytes: artifact.stages.ingestion.rawBytes,
              rawWords: artifact.stages.ingestion.rawWords,
              compressionRatio: artifact.stages.compression.skeleton.ratio,
              chunkCount: artifact.stages.vectorization.chunkCount,
              highValueHits: artifact.stages.compression.manifestEntry.highValueHits,
              fileHash: artifact.stages.compression.manifestEntry.hash,
            });
            processed++;
            if (processed % 10 === 0) {
              console.log(`[pipeline] Processed ${processed} documents...`);
            }
          } catch (docErr: any) {
            console.error(`[pipeline] Error processing "${doc.title}":`, docErr.message);
            skipped++;
          }
        }
        console.log(`[pipeline] Complete: ${processed} processed, ${skipped} skipped, ${docs.length} total`);
      } catch (e: any) {
        console.error("[pipeline] Fatal error:", e.message);
      } finally {
        pipeline.setRunning(false);
      }
    })();
  });
