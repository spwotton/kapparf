! PartyTown Blocker: uBlock Origin Ruleset
! ==============================================
!
! Blocks malicious PartyTown service worker injection and related vectors.
!
! Installation:
!   1. In uBlock Origin settings → My filters (or: My rules)
!   2. Copy-paste these rules
!   3. Apply
!
! Reference: Congusto-Eitel project — airbnb.com.co Service Worker injection

! ===== DOMAIN BLOCKING =====
! Block the anomalous domain entirely
||airbnb.com.co^

! Block PartyTown CDN variants
||partytown.js^
||partytown/^
||ptp-worker.js^

! Block common PartyTown hosting paths
||cdn.jsdelivr.net/npm/partytown/$script
||unpkg.com/partytown/$script
||esm.sh/partytown/$script

! ===== PATH BLOCKING =====
! Block from any domain to these suspicious paths
||*/static/partytown/*
||*/_next/static/partytown/*
||*/lib/partytown/*
||*/js/partytown/*
||*/partytown/lib/*

! ===== SERVICE WORKER BLOCKING =====
! Block service worker registration scripts
||*/service-worker.js?*partytown*
||*/sw.js?*partytown*
||*/worker.js?src=partytown*

! ===== SCRIPT INJECTION PATTERNS =====
! Block inline script tags that load PartyTown
||*/partytown*.js$script
||*/ptp*.js$script

! ===== XHR/FETCH BLOCKING =====
! Block requests to suspicious domains from page context
||airbnb.com.co^$xhr,domain=~airbnb.com.co
||airbnb.com.co^$fetch,domain=~airbnb.com.co

! Block fetch requests with suspicious headers
! (Note: uBlock Origin has limited header inspection)
||*/credential*partytown*
||*/auth*airbnb.com.co*

! ===== BEHAVIOR-BASED BLOCKING =====
! Block scripts that attempt to load into SharedWorker
||*/partytown*$worker

! ===== REDIRECT RULES (uBlock Origin Pro) =====
! If using advanced features, redirect malicious libs to null
! ||partytown^$redirect=noop.js

! ===== COMMENT RULES =====
! Log patterns matching "46.875" or heartbeat anomalies
! (This is for awareness; uBlock cannot directly block by timing frequency)

! ===== WHITELIST EXCEPTIONS =====
! If you need legitimate PartyTown on trusted sites, add here:
! @@||trusted-domain.com^$script,partytown

! ===== DEBUGGING (remove in production) =====
! Enable logging for this filter list
! ! Enable in uBlock → Settings → Advanced → logger
